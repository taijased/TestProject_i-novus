1) Напишите функцию сортировки массива по возрастанию не используя .sort(). Требуется свой алгоритм.

var list = [17, 3, 89, -7, 18, 67.5];

Решение

var list = [17, 3, 89, -7, 18, 67.5];

var count = list.length - 1;

for (var i = 0; i < count; i++)

for (var j = 0; j < count-i; j++)

if (list[j]> list[j+1]) {

var temp = list[j];

list[j] = list[j+1];

list[j+1] = temp;

}

alert(list);
